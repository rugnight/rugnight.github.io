[UE/C++]EnhancedInputをC+＋で扱う

27/09/2024 らぐさん
UnrealEngine
#pragma once
  
#include "CoreMinimal.h"
#include "GameFramework/Character.h"
#include "HogePawn.generated.h"
  
UCLASS()
class CRICKET_API AHogePawn : public ACharacter
{
 GENERATED_BODY()
  
private:
 // 入力セットアップ時に呼ばれる
 virtual void SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent) override;
  
 // Bindする関数
 void OnInvokedAttackAction(const struct FInputActionInstance& Instance);
  
public:
 // BPから指定する入力アクション、BPからアセットを指定する
 UPROPERTY(BlueprintReadOnly, EditAnywhere, Category = "Input")
 const class UInputAction* InputAction;
};
  

#include "HogePawn.h"
#include "InputTriggers.h"
#include "InputAction.h"
#include "EnhancedInputComponent.h"
  
void AHogePawn::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent)
{
 Super::SetupPlayerInputComponent(PlayerInputComponent);
  
 // EnhancedInputへキャスト
 UEnhancedInputComponent* Input = Cast<UEnhancedInputComponent>(PlayerInputComponent);
 
 // アクションに関数をバインド
 Input->BindAction(InputAction, ETriggerEvent::Triggered, this, &AHogePawn::OnInvokedAttackAction);
}
  
  
void AHogePawn::OnInvokedAttackAction(const FInputActionInstance& Instance)
{
     // 入力アクションの値を取得したり
     FVector VectorValue = Instance.GetValue().Get<FVector>();
     float FloatValue = Instance.GetValue().Get<float>();
     bool BoolValue = Instance.GetValue().Get<bool>();

     //  ここで具体的な処理
     GEngine->AddOnScreenDebugMessage(-1, 1.0f, FColor::Yellow, "Input");
}
